axios.get('url/地址').then()
在vue.config.js文件中

方式1
devServer：{
	proxy：'url:端口号'
}

axios.get('url:端口号/name/地址').then()
方式2
vue.config.js文件
devServer：{
	proxy：{
		'/name'：{
			target：'url:端口号',
			pathRewrite:{'/^name/':""},
			ws：true,  // 用于支持websocket
			changeOrigin：true  // 控制请求头中的host值，默认true
		}
	}
}
